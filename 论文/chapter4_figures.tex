\begin{figure}[H]
\centering
\begin{tikzpicture}[

\node[box, fill=blue!10!white, draw=blue!60!black] (client) at (0,0) {浏览器};

\node[box, fill=orange!10!white, draw=orange!60!black] (gateway) at (0,-5) {Gateway\\网关};

\node[box, fill=purple!10!white, draw=purple!60!black] (article) at (0,-7.5) {文章模块};
\node[box, fill=purple!10!white, draw=purple!60!black] (comment) at (3.5,-7.5) {评论模块};

\node[box, fill=teal!10!white, draw=teal!60!black] (mysql) at (0,-10) {MySQL\\数据库};

% 箭头连接
\draw[arrow] (client) -- (frontend) node[right, midway] {HTTP/REST};
\draw[arrow] (frontend) -- (gateway) node[right, midway] {API};

\draw[arrow] (gateway) -- (comment);

\draw[arrow] (article) -- (mysql);
\draw[arrow] (comment) -- (mysql);

% 层标注
\node[layer, fit={(client) (frontend)}] (frontend_layer) {};

\node[layer, fit={(gateway) (user) (article) (comment)}] (backend_layer) {};
\node[above=0.1cm of backend_layer] {服务层};

\node[layer, fit={(mysql)}] (data_layer) {};
\node[above=0.1cm of data_layer] {数据层};

\end{tikzpicture}
